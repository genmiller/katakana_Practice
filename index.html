<script>

(function(){

function hiraToKata(s){

  var out=""; for(var i=0;i<s.length;i++){

    var c=s.charCodeAt(i);

    out += (c>=0x3041 && c<=0x3096) ? String.fromCharCode(c+0x60) : s.charAt(i);

  } return out;

}

var WORDS=["いぬ","ねこ","さかな","とり","うさぎ","ぞう","きつね","たぬき","くま",

"りんご","みかん","ぶどう","いちご","すいか","もも","なし","ばなな",

"やさい","にんじん","たまねぎ","じゃがいも","きゅうり","とまと",

"たまご","ぱん","ごはん","みそしる","おにぎり","おちゃ","みず","ぎゅうにゅう",

"くるま","でんしゃ","ひこうき","ふね","じてんしゃ","ばす","しんごう","みち",

"いえ","へや","まど","どあ","いす","つくえ","ほん","えんぴつ","けしごむ","かばん",

"がっこう","きょうしつ","せんせい","ともだち","えんそく","こうえん",

"あさ","ひる","よる","きのう","きょう","あした",

"はる","なつ","あき","ふゆ","あめ","ゆき","かぜ","たいよう","ほし","つき",

"あたま","かお","め","はな","みみ","くち","て","あし"];

var show=false, cur=[];

var list=document.getElementById("list");

function pick5(){

  var picked=[], used={};

  while(picked.length<5){

    var w=WORDS[Math.floor(Math.random()*WORDS.length)];

    if(!used[w]){ used[w]=1; picked.push(w); }

  } return picked;

}

function render(){

  list.innerHTML="";

  for(var i=0;i<cur.length;i++){

    var li=document.createElement("li");

    li.className="item"+(show?" show":"");

    var a=document.createElement("div"); a.className="hira"; a.textContent=(i+1)+". "+cur[i];

    var b=document.createElement("div"); b.className="ans"; b.textContent=hiraToKata(cur[i]);

    li.appendChild(a); li.appendChild(b); list.appendChild(li);

  }

  document.getElementById("toggle").textContent = show ? "こたえ：みる" : "こたえ：みない";

}

document.getElementById("gen").onclick=function(){ cur=pick5(); render(); };

document.getElementById("toggle").onclick=function(){ show=!show; render(); };

cur=pick5(); render();

})();

</script>
